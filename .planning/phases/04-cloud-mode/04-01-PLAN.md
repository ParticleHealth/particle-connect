  - project = var.project_id
  - location = var.region
  - friendly_name = "Particle Flat Data Observatory"

  - dataset_id = google_bigquery_dataset.observatory.dataset_id
  - project = var.project_id
  - table_id = each.key
  - deletion_protection = false
